--- layout: post title: Git client configuration --- This is the Git configuration I use for my local repositories. 

[core]
 autocrlf = false
[push]
 default = simple
[rerere]
 enabled = true
[diff]
 tool = p4merge
[difftool "p4merge"]
 cmd = p4merge \"$LOCAL\" \"$REMOTE\"
[merge]
 defaultToUpstream = true
 tool = p4merge
[mergetool "p4merge"]
 cmd = p4merge \"$BASE\" \"$LOCAL\" \"$REMOTE\" \"$MERGED\"
 keepTemporaries = false
 trustExitCode = false
 keepBackup = false
[gc]
 auto = 0
